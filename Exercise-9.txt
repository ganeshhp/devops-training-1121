Docker compose file deployed to docker swarm cluster with docker stack.

1) create a docker-compose file and add below contents to the file.

----------------------------------------
version: "3.9"

services:
  web:
    image: nginx:stable
    ports:
      - "80:80"
    deploy:
      mode: replicated
      replicas: 2
      restart_policy:
        condition: on-failure
    networks:
      - frontend

  api:
    image: hashicorp/http-echo:0.2.3
    command: ["-text=Hello from API"]
    deploy:
      mode: replicated
      replicas: 2
      restart_policy:
        condition: on-failure
    networks:
      - frontend

  redis:
    image: redis:alpine
    deploy:
      mode: replicated
      replicas: 1
      restart_policy:
        condition: on-failure
    networks:
      - backend

networks:
  frontend:
    driver: overlay
  backend:
    driver: overlay

----------------------------------------------------

2) $ docker swarm init

3) $ docker stack deploy --compose-file docker-compose.yml mystack

4) $ docker stack services mystack

5) $ docker ps
